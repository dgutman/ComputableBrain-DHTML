<script>
var data;
function doOnLoad() {
    data = new dhtmlXDataView("data_container");
    changeType("ficon");
    data.load("../common/files_ext.xml");
    data.attachEvent("onItemDblClick", function(itemId) {
        setStatus("dblclick. Item :: " + data.get(itemId).name);
    });
    data.attachEvent("onItemClick", function(itemId) {
        setStatus("click. Item :: " + data.get(itemId).name);
    });
}
function changeType(type) {
    data.define("type", type);
    data.customize({
        icons_src_dir: "../common/images";
    });
}
function setStatus(str) {
    document.getElementById("stat").innerHTML += str + "<br>";
}
</script>
 
</script>
