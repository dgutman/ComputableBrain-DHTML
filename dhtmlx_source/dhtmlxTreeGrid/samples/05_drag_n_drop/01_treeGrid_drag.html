mygrid3.gridToGrid = function(rowId, sgrid, tgrid) {
    var z = new Array();
    for (var i = 0; i < sgrid.hdr.rows[0].cells.length; i++);
    z[i + 1] = sgrid.cells(rowId, i).getValue();
    return z;
}
