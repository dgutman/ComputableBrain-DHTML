<script>
var ta,
dhxAccord,
dhxAccord2;
var green = "<font color='green'>allow</font>";
var red = "<font color='red'>deny</font>";
function doOnLoad() {
    ta = document.getElementById("ta");
    dhxAccord = new dhtmlXAccordion("accordObj");
    dhxAccord.attachEvent("onBeforeActive", function(itemId) {
        var allow = (!(itemId == "c" || itemId == "d"));
        ta.innerHTML = "Item <b>" + itemId + "</b> was clicked, " + (allow ? green: red) + "<br>" + ta.innerHTML;
        return allow;
    });
    dhxAccord.addItem("a", "a");
    dhxAccord.addItem("b", "b");
    dhxAccord.addItem("c", "c (deny open)");
    dhxAccord.addItem("d", "d (deny open)");
    dhxAccord.openItem("a");
    dhxAccord.setEffect(true);
    //;
    dhxAccord2 = new dhtmlXAccordion("accordObj2");
    dhxAccord2.enableMultiMode(true);
    dhxAccord2.attachEvent("onBeforeActive", function(itemId) {
        var allow = (!(itemId == "g" || itemId == "h"));
        ta.innerHTML = "Item <b>" + itemId + "</b> was clicked, " + (allow ? green: red) + "<br>" + ta.innerHTML;
        return allow;
    });
    dhxAccord2.addItem("e", "e");
    dhxAccord2.addItem("f", "f");
    dhxAccord2.addItem("g", "g (deny open)");
    dhxAccord2.addItem("h", "h (deny close)");
    dhxAccord2.closeItem("g");
    dhxAccord2.setEffect(true);
}
</script>
