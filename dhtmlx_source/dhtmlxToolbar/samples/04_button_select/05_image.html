<script>
var sel = document.getElementById("sel");
var toolbar = new dhtmlXToolbarObject("toolbarObj");
toolbar.setIconsPath("../common/imgs/");
toolbar.loadXML("../common/dhxtoolbar_buttonselect.xml?etc=" + new Date().getTime(), updateList);
function getId() {
    var id = sel.options[sel.selectedIndex].value;
    return id;
}
var img = "new";
var lastImg = document.getElementById("new");
function changeImage(imgObj) {
    img = imgObj.id;
    imgObj.style.border = "#707070 1px solid";
    lastImg.style.border = "#E0E0E0 1px solid";
    lastImg = imgObj;
}
function clearImage(text) {
    toolbar.clearItemImage(getId());
}
function setImage() {
    toolbar.setItemImage(getId(), img + ".gif");
}
function updateList() {
    sel.options.length = 0;
    toolbar.forEachItem(function(itemId) {
        if (toolbar.getType(itemId) == "buttonSelect") {
            sel.options.add(new Option(itemId, itemId));
        }
    });
}
</script>
