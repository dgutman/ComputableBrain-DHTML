var dhxLayout;
function doOnLoad() {
    dhxLayout = new dhtmlXLayoutObject("parentId", "3L");
    dhxLayout.dhxWins.enableAutoViewport(false);
    dhxLayout.dhxWins.attachViewportTo("winVP");
    dhxLayout.attachEvent("onDock", doOnDock);
    dhxLayout.attachEvent("onUnDock", doOnUnDock);
    dhxLayout.cells("a").attachObject("objA");
    dhxLayout.cells("b").attachObject("objB");
    dhxLayout.cells("c").attachObject("objC");
}
function doOnDock(itemId) {
    document.getElementById("log").innerHTML += "<b>onDock</b> event called, item id is <b>" + itemId + "</b><br>";
}
function doOnUnDock(itemId) {
    document.getElementById("log").innerHTML += "<b>onUnDock</b> event called, item id is <b>" + itemId + "</b><br>";
    dhxLayout.dhxWins.window(itemId).keepInViewport(true);
}
</script>
